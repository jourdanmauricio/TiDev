Clases CSS

Para acceder a los atributos utilizamos document.querySelector('.nombre-clase').atributo; pero si queremos acceder a las clases como atributo tendríamos que utilizar .class, y la palabra class es una palabra reservada en los lenguajes de programación (relacionada a POO).

Entonces, accedemos a través de la propiedad className o a través de un objeto especial que retorna las clases del elemento indicado (propiedad classList).

Para ejemplificar la manipulación de clases utilizaremos algunas tarjetas.

        
          <!-- En la etiqueta Head definmos algunos estilos -->
          <style>
            .card{
              display: inline-block;
            }
            .card figcaption{
                padding: 1rem;
            }
            .rotate-45{
                transform: rotate(45deg);
            }
            .rotate-135{
                transform: rotate(135deg);
            }
            .opacity-80{
                opacity: .8;
            }
            .sepia{
                filter: sepia(1);
            }
        </style>

        <!-- En el body -->
          <section class="cards">
            <figure class="card">
              <img src="https://placeimg.com/200/200/tech" alt="Tech">
              <figcaption>Tech</figcaption>
            </figure>
            <figure class="card">
              <img src="https://placeimg.com/200/200/animals" alt="Animals">
              <figcaption>Animals</figcaption>
            </figure>
            <figure class="card">
              <img src="https://placeimg.com/200/200/arch" alt="Architecture">
              <figcaption>Architecture</figcaption>
            </figure>
          </section>
        
      
        
          /* Seleccionamos la primera de las card */
          const $card = document.querySelector( '.card' );
          console.log( $card );
  
          /* Accediendo al valor del atributo class */$card
          console.log( $card.className ); // card
          console.log( $card.classList );
  
          /* Verificar clase en elemento */
          console.log( $card.classList.contains( 'rotate-45' ) ); // false
  
          /* Agramos clase */
          $card.classList.add( 'rotate-45' );
          console.log( $card.classList.contains( 'rotate-45' ) ); // true
          console.log( $card.classList );
  
          /* Removiendo una clase */
          $card.classList.remove( 'rotate-45' );
          console.log( $card.classList.contains( 'rotate-45' ) ); // false
  
          /* Toggle - Palanca. Funciona a menara de interruptor, 
          si tiene la clase la quita, si no la tiene la agrega */
          $card.classList.toggle( 'rotate-45' );
          console.log( $card.classList.contains( 'rotate-45' ) ); // true
          $card.classList.toggle( 'rotate-45' );
          console.log( $card.classList.contains( 'rotate-45' ) ); // false
  
          /* Reemplazando clases */
          $card.classList.toggle( 'rotate-45' );
          $card.classList.replace( 'rotate-45', 'rotate-135' );
  
          /* Trabajando varias clases de una vez */
          $card.classList.add( 'opacity-80', 'sepia' );
          $card.classList.remove( 'opacity-80', 'sepia' );
          $card.classList.toggle( 'opacity-80', 'sepia' );
